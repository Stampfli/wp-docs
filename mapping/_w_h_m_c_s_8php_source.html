456  private function _populate_dns_records( $tld, $sld ) {

460  $options = Domainmap_Plugin::instance()->get_options();
461 
462  // if server ip addresses are provided, use it to populate DNS records
463  if ( !empty( $options['map_ipaddress'] ) ) {
464  foreach ( explode( ',', trim( $options['map_ipaddress'] ) ) as $ip ) {
465  if ( filter_var( $ip, FILTER_VALIDATE_IP ) ) {
466  $ips[] = $ip;
467  }
468  }
469  }
470 
471  // looks like server ip addresses are not set, then try to read it automatically
472  if ( empty( $ips ) && function_exists( 'dns_get_record' ) ) {
473  // fetch unchanged domain name from database, because get_option function could return mapped domain name
474  $basedomain = parse_url( $wpdb->get_var( "SELECT option_value FROM {$wpdb->options} WHERE option_name = 'siteurl'" ), PHP_URL_HOST );
475  // fetch domain DNS A records
476  $dns = @dns_get_record( $basedomain, DNS_A );
477  if ( is_array( $dns ) ) {
478  $ips = wp_list_pluck( $dns, 'ip' );

480  }
481

483  $dedicated = false;
484  if ( !empty( $ips ) ) {
485  $check = sha1( time() );
486 
487  switch_to_blog( 1 );
488  $ajax_url = admin_url( 'admin-ajax.php' );
489  $ajax_url = str_replace( parse_url( $ajax_url, PHP_URL_HOST ), current( $ips ), $ajax_url );
490  restore_current_blog();
491 
492  $response = wp_remote_request( add_query_arg( array(
493  'action' => Domainmap_Plugin::ACTION_HEARTBEAT_CHECK,
494  'check' => $check,
495  ), $ajax_url ) );
496 
497  $dedicated = !is_wp_error( $response ) && wp_remote_retrieve_response_code( $response ) == 200 && wp_remote_retrieve_body( $response ) == $check;
498  }
499 
500  // populate request arguments
501  if ( !empty( $ips ) && $dedicated ) {
502  // network is hosted on dedicated hosting and we can use DNS A records
503  $i = 0;
504  foreach ( $ips as $ip ) {
505  if ( filter_var( trim( $ip ), FILTER_VALIDATE_IP ) ) {
506  $i++;
507  $args["HostName{$i}"] = '@';
508  $args["RecordType{$i}"] = 'A';
509  $args["Address{$i}"] = $ip;
510  }
511  }
512  } else {
513  // network is hosted on shared hosting and we can use DNS CNAME records for it
514  $origin = $wpdb->get_row( "SELECT * FROM {$wpdb->blogs} WHERE blog_id = " . intval( $blog_id ) );
515

522  $args['Address2'] = "{$origin->domain}.";
523  }
524

527  $args['sld'] = $sld;
528  $args['tld'] = $tld;
529
