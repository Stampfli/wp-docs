559  function is_prohibited_domain( $domain, $check_subdomains = true ){
560  $probibited_domains = $this->get_option("map_prohibited_domains");
561  $probibited_domains = empty( $probibited_domains ) ? array() : explode(",", $probibited_domains ) ;
562 
563  if( !count( $probibited_domains ) ) return false;
564 
565  $probibited_domains = array_map('trim',$probibited_domains);
566  if( $check_subdomains ){
567  foreach( $probibited_domains as $probibited_domain){
568  if( $domain === $probibited_domain || strpos( $domain, "." . $probibited_domain ) !== false )
569  return true;
570  }
571  }
572 
573  return in_array($domain, $probibited_domains);
574  }
