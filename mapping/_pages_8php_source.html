103  public function render_site_options_page() {

120  $page = new Domainmap_Render_Site_Purchase( $tabs, $activetab, $options );
121  $page->reseller = $reseller;

123  // fetch unchanged domain name from database, because get_option function could return mapped domain name

130  $ips[] = $ip;
131  }
132  }
133  }
134 
135  // looks like server ip addresses are not set, then try to read it automatically
136  if ( empty( $ips ) && function_exists( 'dns_get_record' ) ) {
137  $dns = @dns_get_record( $basedomain, DNS_A );
138  if ( is_array( $dns ) ) {
139  $ips = wp_list_pluck( $dns, 'ip' );
140  }
141  }
142 
143  // prepare template
144  $page = new Domainmap_Render_Site_Map( $tabs, $activetab, $options );
145  $page->origin = $this->_wpdb->get_row( "SELECT * FROM {$this->_wpdb->blogs} WHERE blog_id = " . intval( $blog_id ) );
146  $page->domains = (array)$this->_wpdb->get_results( sprintf( "SELECT domain, is_primary FROM %s WHERE blog_id = %d ORDER BY id ASC", DOMAINMAP_TABLE_MAP, (int) $blog_id ) );

313  $items = array_filter( array_map( 'intval', $items ) );
