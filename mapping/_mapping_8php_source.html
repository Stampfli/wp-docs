446  private static function _build_url( $components ) {
447  $scheme = isset( $components['scheme'] ) ? $components['scheme'] . '://' : '';
448  $host = isset( $components['host'] ) ? $components['host'] : '';
449  $port = isset( $components['port'] ) ? ':' . $components['port'] : '';

458  return $scheme . str_replace("//", "/", $user . $pass . $host . $port . $path . $query . $fragment );
