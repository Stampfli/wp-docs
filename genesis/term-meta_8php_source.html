209 function genesis_get_term_filter( $term, $taxonomy ) {
210 
211  //* Do nothing, if $term is not object
212  if ( ! is_object( $term ) ) {
213  return $term;
214  }
215 
216  //* Do nothing, if called in the context of creating a term via an ajax call
217  if ( did_action( 'wp_ajax_add-tag' ) ) {
218  return $term;
219  }
220 
221 
222  $db = get_option( 'genesis-term-meta' );
223  $term_meta = isset( $db[$term->term_id] ) ? $db[$term->term_id] : array();
224 
225  $term->meta = wp_parse_args( $term_meta, apply_filters( 'genesis_term_meta_defaults', array(
226  'headline' => '',
227  'intro_text' => '',
228  'display_title' => 0, //* vestigial
229  'display_description' => 0, //* vestigial
230  'doctitle' => '',
231  'description' => '',
232  'keywords' => '',
233  'layout' => '',
234  'noindex' => 0,
235  'nofollow' => 0,
236  'noarchive' => 0,
237  ) ) );
238 
239  //* Sanitize term meta
240  foreach ( $term->meta as $field => $value )
241  $term->meta[$field] = apply_filters( "genesis_term_meta_{$field}", stripslashes( wp_kses_decode_entities( $value ) ), $term, $taxonomy );
242 
243  $term->meta = apply_filters( 'genesis_term_meta', $term->meta, $term, $taxonomy );
244 
245  return $term;
246 
247 }
