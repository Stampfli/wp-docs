127  toggleSettings: function( event ) {
128  'use strict';
129 
130  // Cache selectors
131  var $selector = jQuery( event.data.selector ),
132  $showSelector = jQuery( event.data.showSelector ),
133  checkValue = event.data.checkValue;
134 
135  // Compare if a checkValue is an array, and one of them matches the value of the selected option
136  // OR the checkValue is _unchecked, but the checkbox is not marked
137  // OR the checkValue is _checked, but the checkbox is marked
138  // OR it's a string, and that matches the value of the selected option.
139  if (
140  ( jQuery.isArray( checkValue ) && jQuery.inArray( $selector.val(), checkValue ) > -1) ||
141  ( '_unchecked' === checkValue && $selector.is( ':not(:checked)' ) ) ||
142  ( '_checked' === checkValue && $selector.is( ':checked' ) ) ||
143  ( '_unchecked' !== checkValue && '_checked' !== checkValue && $selector.val() === checkValue )
144  ) {
145  jQuery( $showSelector ).slideDown( 'fast' );
146  } else {
147  jQuery( $showSelector ).slideUp( 'fast' );
